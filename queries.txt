// Table operations

CREATE TABLE studs {id: INTEGER, name:VARCHAR(32), age: INTEGER};
REMOVE TABLE studs;

// Inserts

INSERT {id:5, name:'Stefan', age:18} IN studs;

FOR stud IN
[ {id:1, name:'Stefan', age:22},
  {id:2, name:'Vasil', age:23},
  {id:3, name:'Vitali', age:18}
  ] INSERT IN studs;

// Updates

UPDATE
    WITH {id:1, name:'Stefan', age:22}
    IN studs;

FOR stud IN studs
    FILTER stud.id > 0 AND stud.id < 10
    FILTER 15 > 0
    UPDATE stud
        WITH {id:1, name:'Stefan', age:22}
        IN studs;

// Removals

REMOVE IN studs;

FOR stud in studs
   FILTER stud.id > 0
   REMOVE stud in studs;

FOR stud in studs
   FILTER stud.id > 0 AND stud.id < 10
   REMOVE stud in studs;

FOR stud in studs
   FILTER stud.id > 0 AND stud.id < 10
   FILTER stud.age < 20
   REMOVE stud in studs;

// Returns

FOR stud IN studs
    RETURN stud;

FOR stud IN studs
    FILTER stud.id > 0 AND stud.id < 10
    FILTER stud.age < 20
    RETURN stud;

FOR stud IN studs
    RETURN { id, age, name };

FOR stud IN studs
    FILTER stud.id > 0 AND stud.id < 10
    FILTER stud.age < 20
    RETURN { id, age, name };

FOR stud IN studs
    FOR prof IN profs
        FILTER stud.id = prof.id
        RETURN stud, prof;

FOR stud IN studs
    FOR prof IN profs
        FILTER stud.id = prof.id
        FILTER stud.id > 0 AND stud.id < 10
        FILTER prof.age > 40
        RETURN stud, prof;

FOR stud IN studs
    FOR prof IN profs
        FILTER stud.id = prof.id
        RETURN { id, age, name };

FOR stud IN studs
    FOR prof IN profs
        FILTER stud.id = prof.id
        FILTER stud.id > 0 AND 'stud.id' < 'stefan'
        FILTER stud.age < 20
        RETURN { id, age, name };
